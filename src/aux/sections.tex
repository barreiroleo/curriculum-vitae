% input: 1(opt) left width, 2 right width, 3 left content, 4 right content
\newcommand{\SectionBase}[4][1=1.8cm]{
	\adjustbox{valign=t}{
		\begin{minipage}{#1}
			{#3}
		\end{minipage}
	}
	\vline \hspace{10pt}
	\adjustbox{valign=t}{
		\begin{minipage}{#2}
			{#4}
		\end{minipage}
	}
}

% WORK EXPERIENCE
% Input: 1 right width, 2 year, 3 position, 4 location, 5 Company, 6 Description
\newcommand{\WorkBase}[6]{ 
	\SectionBase[1.8cm]{#1}
	{ \center \small \bfseries #2\\ }
	{
		{\normalsize \bfseries #3}\hfill
		{\footnotesize\faMapMarker~~#4}\newline
		\small {\emph{#5}}\\
		\footnotesize {#6}
	}
	\vspace{-5pt}
}
\newcommand{\WorkMiniWidth}[5]{
	\WorkBase{0.80\textwidth}{#1}{#2}{#3}{#4}{#5}
}
\newcommand{\WorkFullWidth}[5]{
	\WorkBase{0.85\textwidth}{#1}{#2}{#3}{#4}{#5}
}

% ACADEMIC: 1 title, 2 year, 3 content
\newcommand{\Academic}[3]{
	\begin{itemize}
		\item \textbf{#1} \textit{#2} \newline \footnotesize {#3}
	\end{itemize}
	\vspace{-20pt}
}

% ITC section: 1 branch, 2 description
\newcommand{\Tech}[2]{
	\SectionBase[3.0cm]{0.80\linewidth}
	{
		\begin{flushright}
			{\small \bfseries #1}
		\end{flushright}
	}
	{\footnotesize #2}
}
