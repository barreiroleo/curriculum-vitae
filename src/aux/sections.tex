\newcommand{\FullWidth}[0]{0.88\linewidth}
\newcommand{\MiniWidth}[0]{0.85\linewidth}

% input: 1(opt) left width, 2 right width, 3 left content, 4 right content
\newcommand{\SectionBase}[4][1.5cm]{
	\adjustbox{valign=t}{
		\begin{minipage}{#1}
			{#3}
		\end{minipage}
	}
	\vline \hspace{3pt}
	\adjustbox{valign=t}{
		\begin{minipage}{#2}
			{#4}
		\end{minipage}
	}
}
% ListItem 1 Left margin, 2 title, 3 year, 4 content
\usepackage{enumitem}
\newcommand{\ListItem}[4][7pt]{
	\begin{itemize}[leftmargin=#1]
		\item \textbf{#2} \textit{#3} \newline \footnotesize {#4}
	\end{itemize}
	\vspace{-20pt}
}

% EDUCATION
% Input: 1[opt] year, 2 logo, 3 content
\newcommand{\Study}[7]{ % input: years, logo, title, university, city, specialization, description
	\SectionBase[1.8cm]{\MiniWidth-5ex}{
		\begin{center}
			{\small \bfseries #1} \\ \vspace*{0.2cm}
			\includegraphics[width=1.3cm,height=1.3cm,keepaspectratio]{#2}
		\end{center}
	}{
		{\normalsize\bfseries #3} \hfill {\footnotesize \faMapMarker  ~ #5} \newline
		{\small\bfseries #4} \newline
		\small {\emph{#6}} \newline
		\footnotesize {#7}
	}
}

% WORK EXPERIENCE
% Input: 1 right width, 2 year, 3 position, 4 location, 5 Company, 6 Description
\newcommand{\WorkBase}[6]{
	\SectionBase[1.0cm]{#1}
	{ \center \small \bfseries #2\\ }
	{
		{\normalsize \bfseries #3}\hfill
		{\footnotesize\faMapMarker~~#4}\newline
		\small {\emph{#5}}\\
		\footnotesize {#6}
	}
	\vspace{-5pt}
}
\newcommand{\WorkMiniWidth}[5]{
	\WorkBase{\MiniWidth}{#1}{#2}{#3}{#4}{#5}
}
\newcommand{\WorkFullWidth}[5]{
	\WorkBase{\FullWidth+2ex}{#1}{#2}{#3}{#4}{#5}
}

% ITC section: 1 branch, 2 description
\newcommand{\Tech}[2]{
	\SectionBase[3.0cm]{\FullWidth}
	{
		\begin{flushright}
			{\small \bfseries #1}
		\end{flushright}
	}
	{\footnotesize #2}
}

% ACADEMIC: 1 title, 2 year, 3 content
\newcommand{\Academic}[3]{
	\ListItem[20pt]{#1}{#2}{#3}
}

% Certificates
% Input: 1[opt] institution, 2 logo, 3 content
\newcommand{\Certficate}[3][]{
	\SectionBase{\FullWidth}{
		\begin{center}
			{\small \bfseries #1}
			\vspace*{0.1cm}
			\includegraphics[width=1.2cm,height=1.2cm,keepaspectratio]{#2}
		\end{center}
	}
	{ #3 }
}
