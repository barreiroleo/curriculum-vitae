% input: 1(opt) left width, 2 right width, 3 left content, 4 right content
\newcommand{\SectionBase}[4][1=1.8cm]{
	\adjustbox{valign=t}{
		\begin{minipage}{#1}
			{#3}
		\end{minipage}
	}
	\vline \hspace{10pt}
	\adjustbox{valign=t}{
		\begin{minipage}{#2}
			{#4}
		\end{minipage}
	}
}

% WORK EXPERIENCE
\newcommand{\WorkBase}[6]{ % input: 1 year, 2 position, 3 location, 4 Company, 5 Description
	\SectionBase[1.8cm]{#6}
	{\center \small \bfseries #1\\}
	{
		{\normalsize \bfseries #2}\hfill
		{\footnotesize\faMapMarker~~#3}\newline
		\small {\emph{#4}}\\
		\footnotesize {#5}
	}
	\vspace{-5pt}
}
\newcommand{\WorkMiniWidth}[5]{
	\WorkBase{#1}{#2}{#3}{#4}{#5}{0.80\textwidth}
}
\newcommand{\WorkFullWidth}[5]{
	\WorkBase{#1}{#2}{#3}{#4}{#5}{0.85\textwidth}
}

% ACADEMIC
\newcommand{\Academic}[3]{ %Input: title, year, content
	\begin{itemize}
		\item \textbf{#1} \textit{#2} \newline \small {#3}
	\end{itemize}
	\vspace{-20pt}
}

% ITC section
\newcommand{\ITCcompetence}[2]{ % input: Branch, description
	\SectionBase[3.0cm]{0.80\linewidth}
	{
		\begin{flushright}
			{\footnotesize \bfseries #1}
		\end{flushright}
	}
	{\footnotesize #2}
}
